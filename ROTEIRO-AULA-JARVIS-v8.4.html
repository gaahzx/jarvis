function searchVaultContext(query) {
  const vaultPath = process.env.OBSIDIAN_VAULT_PATH;
  if (!vaultPath || !fs.existsSync(vaultPath)) return '';
  const files = fs.readdirSync(vaultPath).filter(f => f.endsWith('.md'));
  const kw = query.toLowerCase().split(' ');
  const hits = files.map(f => {
    const content = fs.readFileSync(path.join(vaultPath, f), 'utf8');
    const score = kw.filter(k => content.toLowerCase().includes(k)).length;
    return { f, content, score };
  }).filter(x => x.score > 0).sort((a,b) => b.score - a.score).slice(0, 3);
  return hits.map(h => `[${h.f}]\n${h.content.slice(0,400)}`).join('\n\n');
}
